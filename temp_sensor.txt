# reset
i2ctransfer -y 0 w1@0x77 0x1E

# read constants
i2ctransfer -y 0 w1@0x77 0xA0 r2@0x77
i2ctransfer -y 0 w1@0x77 0xA2 r2@0x77
i2ctransfer -y 0 w1@0x77 0xA4 r2@0x77
i2ctransfer -y 0 w1@0x77 0xA6 r2@0x77
i2ctransfer -y 0 w1@0x77 0xA8 r2@0x77
i2ctransfer -y 0 w1@0x77 0xAA r2@0x77

# start conversion
i2ctransfer -y 0 w1@0x77 0x48

# get conversion
i2ctransfer -y 0 w1@0x77 0x00 r3@0x77

# get sn
i2ctransfer -y 0 w1@0x77 0xAC r2@0x77
i2ctransfer -y 0 w1@0x77 0xAE r2@0x77
